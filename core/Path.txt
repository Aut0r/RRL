function NULL_PATH: TPath;
var
  NullPath: TPath;
  TileArray: TRSTileArray;
  HandlerArray: TIntegerArray;
begin
  SetLength(TileArray, 0);
  SetLength(HandlerArray, 0);
  with NullPath do
  begin
    Tiles := TileArray;
    Handlers := HandlerArray;
  end;
  Result := NullPath;
end;

function r_PathStart(const ThePath: TPath): TRSTile;
begin
  Result := NULL_RSTile;
  if (Length(ThePath.Tiles) > 0) then
    Result := ThePath.Tiles[0];
end;

function r_PathEnd(const ThePath: TPath): TRSTile;
begin
  Result := NULL_RSTile;
  if (Length(ThePath.Tiles) > 0) then
    Result := ThePath.Tiles[Length(ThePath.Tiles) - 1];
end;

function r_PathNextTileEx(const ThePath: TPath; const Dist: Integer): TRSTile;
var
  i, l: Integer;
begin
  Result := NULL_RSTile;
  l := Length(ThePath.Tiles);
  if (l > 0) then
    for i := 0 to l - 1 do
      if (r_TileOnMinimapEx(ThePath.Tiles[i], Dist)) then
        Result := ThePath.Tiles[i];
end;

function r_PathNextTile(const ThePath: TPath): TRSTile;
var
  i, l: Integer;
begin
  Result := NULL_RSTile;
  l := Length(ThePath.Tiles);
  if (l > 0) then
    for i := 0 to l - 1 do
      if (r_TileOnMinimap(ThePath.Tiles[i])) then
        Result := ThePath.Tiles[i];
end;

function r_PathAtDestination(const ThePath: TPath): Boolean;
begin
  Result := r_DistanceBetween(r_GetLocation, r_PathEnd(ThePath)) < 3;
end;

function r_PathStep(const ThePath: TPath): Boolean;
var
  OurTile, TheClickTile, FlagTile: TRSTile;
  l, r, rr: Integer;
  Params: TIntegerArray;
begin
  l := Length(ThePath.Tiles);
  TheClickTile := NULL_RSTILE;
  if (l > 0) then
  begin
    OurTile := r_GetLocation;
    if (r_DistanceBetween(OurTile, r_PathEnd(ThePath)) < 3) then
    begin
      Result := True;
      Exit;
    end;
    r := RandomRange(14, 17);
    TheClickTile := r_PathNextTileEx(ThePath, r);
  end;
  if (not (TheClickTile = NULL_RSTILE)) then
  begin
    Params := ThePath.Handlers;
    if (InIntArray(Params, PATH_HANDLER_RUN)) then
    begin
      //TODO .
    end;
    if (InIntArray(Params, PATH_HANDLER_WALKING)) then
    begin
      FlagTile := r_GetFlagPosition;
      if (FlagTile.x < 0) then
        FlagTile := NULL_RSTILE;
      if (FlagTile = NULL_RSTILE) then
        r_ClickTile(TheClickTile)
      else
      begin
        rr := RandomRange(4, 8);
        if (r_DistanceBetween(r_GetLocation, FlagTile) <= rr) then
          r_ClickTile(TheClickTile);
      end;
    end else
      r_ClickTile(TheClickTile);
    Result := True;
    Exit;
  end;
  Result := False;
end;

function r_PathReverse(const ThePath: TRSTileArray): TRSTileArray;
var
  i, l: Integer;
begin
  l := Length(ThePath);
  SetLength(Result, l);
  for i := 0 to l - 1 do
    Result[l - i - 1] := ThePath[i];
end;
